#include<avr/io.h>
#include<util/delay.h>
void cmd(unsigned char x)
{
PORTD&=(~0X80);
PORTD&=(~0X40);
PORTB=x;
PORTD|=(0X20);
_delay_ms(100);
PORTD&=(~0X20);
}

void dat(unsigned char x)
{
PORTD|=(0X80);
PORTD&=(~0X40);
PORTB=x;
PORTD|=(0X20);
_delay_ms(100);
PORTD&=(~0X20);

}

{
unsigned int  y ,z =1;

}

}
